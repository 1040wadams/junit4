17  import org.junit.runners.model.FrameworkMethod;
18  import org.junit.validator.AnnotationValidator;
19  
20  /**
21   * Validates that there are no errors in the use of the {@code Category}
22   * annotation. If there is, a {@code Throwable} object will be added to the list
23   * of errors.
24   *
25   * @since 4.12
26   */
27  public final class CategoryValidator extends AnnotationValidator {
28  
29      @SuppressWarnings("unchecked")
30      private static final Set<Class<? extends Annotation>> INCOMPATIBLE_ANNOTATIONS = unmodifiableSet(new HashSet<Class<? extends Annotation>>(
31              asList(BeforeClass.class, AfterClass.class, Before.class, After.class)));
32  
33      /**
34       * Adds to {@code errors} a throwable for each problem detected. Looks for
35       * {@code BeforeClass}, {@code AfterClass}, {@code Before} and {@code After}
36       * annotations.
37       *
38       * @param method the method that is being validated
39       * @return A list of exceptions detected
40       *
41       * @since 4.12
42       */
43      @Override
44      public List<Exception> validateAnnotatedMethod(FrameworkMethod method) {
45          List<Exception> errors = new ArrayList<Exception>();
46          Annotation[] annotations = method.getAnnotations();
47          for (Annotation annotation : annotations) {
48              for (Class<?> clazz : INCOMPATIBLE_ANNOTATIONS) {
49                  if (annotation.annotationType().isAssignableFrom(clazz)) {
50                      addErrorMessage(errors, clazz);
51                  }
52              }
53          }
54          return unmodifiableList(errors);
55      }
56  
57      private void addErrorMessage(List<Exception> errors, Class<?> clazz) {
58          String message = String.format("@%s can not be combined with @Category",
59                  clazz.getSimpleName());
60          errors.add(new Exception(message));
61      }
62  }
