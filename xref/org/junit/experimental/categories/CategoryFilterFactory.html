1   package org.junit.experimental.categories;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   
6   import org.junit.internal.Classes;
7   import org.junit.runner.FilterFactory;
8   import org.junit.runner.FilterFactoryParams;
9   import org.junit.runner.manipulation.Filter;
10  
11  /**
12   * Implementation of FilterFactory for Category filtering.
13   */
14  abstract class CategoryFilterFactory implements FilterFactory {
15      /**
16       * Creates a {@link org.junit.experimental.categories.Categories.CategoryFilter} given a
17       * {@link FilterFactoryParams} argument.
18       *
19       * @param params Parameters needed to create the {@link Filter}
20       */
21      public Filter createFilter(FilterFactoryParams params) throws FilterNotCreatedException {
22          try {
23              return createFilter(parseCategories(params.getArgs()));
24          } catch (ClassNotFoundException e) {
25              throw new FilterNotCreatedException(e);
26          }
27      }
28  
29      /**
30       * Creates a {@link org.junit.experimental.categories.Categories.CategoryFilter} given an array of classes.
31       *
32       * @param categories Category classes.
33       */
34      protected abstract Filter createFilter(List<Class<?>> categories);
35  
36      private List<Class<?>> parseCategories(String categories) throws ClassNotFoundException {
37          List<Class<?>> categoryClasses = new ArrayList<Class<?>>();
38  
39          for (String category : categories.split(",")) {
40              Class<?> categoryClass = Classes.getClass(category);
41  
42              categoryClasses.add(categoryClass);
43          }
44  
45          return categoryClasses;
46      }
47  }
